\begin{example}
    设 \( f: \mathbb{R}^n \to \mathbb{R} \) 为正定二次型 \( f(x) = x^\top Q x \)，多变量正态分布为：
    \[
    p(x|\mu, \Sigma) = \frac{1}{(2\pi)^{\frac{n}{2}} |\Sigma|^{\frac{1}{2}}} \exp \left( -\frac{1}{2}(x - \mu)^\top \Sigma^{-1}(x - \mu) \right),
    \]
    计算积分：
    \[
    J(\mu, \Sigma) = \int_{\mathbb{R}^n} f(x) p(x|\mu, \Sigma) \, dx.
    \]
    \end{example}
